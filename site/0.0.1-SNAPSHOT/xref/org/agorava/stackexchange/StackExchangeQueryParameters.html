187     /**
188      * Generates a map of parameters from this object fields values.
189      * @return a map linking parameter names to their values
190      */
191     public Map<String, Object> toMap() {
192         Map<String, Object> returned = new TreeMap<String, Object>();
193         // being a lazy guy, i use reflection instead of a bunch of ifs ... bad ? maybe, but efficient like hell !
194         for(Field f : getClass().getDeclaredFields()) {
195             if(!Modifier.isStatic(f.getModifiers()) && !Modifier.isFinal(f.getModifiers())) {
196                 try {
197                     Object value = f.get(this);
198                     if(value!=null && !value.equals(0))
199                         returned.put(f.getName(), value);
200                 } catch (IllegalArgumentException ex) {
201                     Logger.getLogger(StackExchangeQueryParameters.class.getName()).log(Level.SEVERE, null, ex);
202                 } catch (IllegalAccessException ex) {
203                     Logger.getLogger(StackExchangeQueryParameters.class.getName()).log(Level.SEVERE, null, ex);
204                 }
205             }
206         }
207         return returned;
208     }
